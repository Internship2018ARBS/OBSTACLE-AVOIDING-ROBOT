#define BUZZER_PIN A3

#define GREEN_LED A2
#define GREEN_LED_DELAY 1000
bool GREEN_LED_STATE = true;
bool BUZZER_STATE = false;
unsigned long GREEN_LED_TIMER = 0;


unsigned long BUZZER_TIMER = 0;
unsigned long BUZZER_DELAY = 250;
unsigned long currentMillis;

void buzzer_buzz(){
  if(BUZZER_DELAY != -1){
    currentMillis = millis();
    if(currentMillis - BUZZER_TIMER >= BUZZER_DELAY){
     
            if(BUZZER_STATE){
                 digitalWrite(BUZZER_PIN , LOW);
                 BUZZER_STATE = false;
              
            } else {
              digitalWrite(BUZZER_PIN , HIGH);
                BUZZER_STATE = true;
              
            }
            BUZZER_TIMER = currentMillis;
           
      }
    } else {
      digitalWrite(BUZZER_PIN , LOW);
    }

}

void blink_green_led(){
   currentMillis = millis();
  if(currentMillis - GREEN_LED_TIMER >= GREEN_LED_DELAY){
   
          if(GREEN_LED_STATE){
               GREEN_LED_STATE = false;
            
          } else {
            GREEN_LED_STATE = true;
            
            
          }
          GREEN_LED_TIMER = currentMillis;
         
    }

    if(GREEN_LED_STATE){
         {
              analogWrite(GREEN_LED , 255);
             
         }
    } else {
        analogWrite(GREEN_LED , 0);
       
    } 
}

void go(int spd1, int spd2)
{
  if(spd1>=0)
  {
    digitalWrite(13,LOW);
    digitalWrite(4,HIGH);
    if(spd2>=0)
    {
       digitalWrite(7,LOW);
       digitalWrite(8,HIGH);
       analogWrite(PWM1 , spd1);
       analogWrite(PWM2 , spd2);
    } 
    else
    {
      digitalWrite(7,HIGH);
      digitalWrite(8,LOW);
      analogWrite(PWM1 , spd1);
      analogWrite(PWM2 , -spd2);
    }
  } 
  else 
  {
      digitalWrite(13,HIGH);
      digitalWrite(4,LOW);
      if(spd2>=0)
      {
         digitalWrite(7,LOW);
         digitalWrite(8,HIGH);
         analogWrite(PWM1 , -spd1);
         analogWrite(PWM2 , spd2);
      }
      else 
      {
         digitalWrite(7,HIGH);
         digitalWrite(8,LOW);
         analogWrite(PWM1 , -spd1);
         analogWrite(PWM2 , -spd2);
      }
    }
  }



void setup() {
  pinMode(TRIG_PIN , OUTPUT);
  pinMode(ECHO_PIN , INPUT);
  pinMode(3,OUTPUT);      //pwm motor1
  pinMode(11,OUTPUT);      //pwm motor2
  pinMode(13,OUTPUT);            //for motor1
  pinMode(4,OUTPUT);      //for motor1
  pinMode(7,OUTPUT);      //for motor2
  pinMode(8,OUTPUT);      //for motor2
  pinMode(A0,INPUT_PULLUP);
  pinMode(A5,INPUT_PULLUP);
    pinMode(A3,OUTPUT);
  
  irrecv.enableIRIn();   // REMOTE initializing pin

  srv.attach(SERVO_PIN);
  srv.write(85);
  srv.detach();

 // Serial.begin(9600);

}



int getDistance(){
  digitalWrite(TRIG_PIN , LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN , HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN , LOW);
  duration = pulseIn(ECHO_PIN , HIGH);
  distance = duration * 0.034 / 2;
  return distance;
}


unsigned long stanga_135 , dreapta_45;
int  searchLeftandRight(int pin){
    digitalWrite(A3 , LOW);
    srv.attach(pin);
    
    srv.write(0);
    delay(1000);
    dreapta = getDistance();

   srv.write(45);
    delay(1000);
    dreapta_45 = getDistance();

      srv.write(135);
    delay(1000);
    stanga_135 = getDistance();
    
    srv.write(180);
    delay(1000);  
    stanga = getDistance();

   // revine la pozitia initiala
    srv.write(85);
    delay(1000);
    
    srv.detach();

      if(dreapta < dreapta_45 && dreapta < stanga_135 && dreapta < stanga){
          return 0;
      }

      if(stanga < dreapta_45 && stanga < stanga_135 && stanga < dreapta){
          return 1;
      }

       if(stanga_135 < dreapta_45 && stanga_135 < stanga && stanga_135 < dreapta){
          return 2;
      }

       if(dreapta_45 < stanga_135 && dreapta_45 < stanga && dreapta_45 < dreapta){
          return 3;
      }

   
}

unsigned long distance_from_obstacle;
int rotation_degrees;
void autonomousRoutine()
{
      distance_from_obstacle = getDistance();
    if(distance_from_obstacle <= 30 || digitalRead(A0) == LOW || digitalRead(A5) == LOW){ //25 cm
           
            go(0,0);
            delay(500);
            // 0 dreapta 1 stanga 2 stanga_135 3 dreapta_45
          rotation_degrees =  searchLeftandRight(SERVO_PIN);
            if(rotation_degrees == 0)
            {
               //Trebuie sa se intoarca spre stanga
               // Serial.println("==== STANGA =====");
                go(-motor1_speed,motor2_speed);
                delay(400);
                go(0,0);
                delay(400);
            }
            else if(rotation_degrees == 1)
            {
            
              
                 //Trebuie sa se intoarca spre dreapta
              // Serial.println("==== DREAPTA =====");
              go(motor1_speed,-motor2_speed);
              delay(600);
              go(0,0);
              delay(600);
            } else if (rotation_degrees == 2){
              //Trebuie sa se intoarca spre dreapta
              // Serial.println("==== DREAPTA =====");
              go(motor1_speed,-motor2_speed);
              delay(300);
              go(0,0);
              delay(300);
            } else {
              
               //Trebuie sa se intoarca spre stanga
               // Serial.println("==== STANGA =====");
                go(-motor1_speed,motor2_speed);
                delay(200);
                go(0,0);
                delay(200);
            }
       } 
       else
       {
            //Merge in fata pana intalneste obstacol
             go(motor1_speed,motor2_speed);
       }
}
